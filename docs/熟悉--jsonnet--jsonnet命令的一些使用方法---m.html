-m


jsonnet -m <dir>


执行时，这会将生成的 JSON 写入给定目录中

[root@centos7 jsonnet_learn]# cat test.jsonnet          
{ 
  "a.json": { 
    x: 1, 
    y: $["b.json"].y, 
  }, 
  "b.json": { 
    x: $["a.json"]. x, 
    y: 2, 
  }, 
}
[root@centos7 jsonnet_learn]# jsonnet test.jsonnet -m ./
./a.json
./b.json
[root@centos7 jsonnet_learn]# cat a.json                
{
   "x": 1,
   "y": 2
}
[root@centos7 jsonnet_learn]# cat b.json                
{
   "x": 1,
   "y": 2
}
