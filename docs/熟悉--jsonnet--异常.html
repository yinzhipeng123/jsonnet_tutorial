抛出异常： error "foo"

下面的程序，定义数组不允许为空，否则异常
[root@centos7 jsonnet_learn]# cat test.jsonnet 
local equal_parts(size, ingredients) =
  local qty = size / std.length(ingredients);
  if std.length(ingredients) == 0 then
    error 'Empty ingredients.'
  else [
    { kind: i, qty: qty }
    for i in ingredients
  ];


{
test1: equal_parts(16, ['Whiskey8',"ooo"]),
test2: equal_parts(1, ['1']),
}

该例子定义a必须大于b

local subtract(a, b) =

assert后面可以加一些判断函数

这些函数在https://jsonnet.org/ref/stdlib.html 能查到


下面的例子，assert std.isObject(self.object), 判断执行的是object才可以执行 ，不是的话就报错

[root@centos7 jsonnet_learn]# cat test.jsonnet 
local subtract(a, b) =
  assert a > b : 'a must be bigger than b';
  a - b;


{
  test2: subtract(10, 3),
  object: {
    assert self.f < self.g : 'wat',
    f: 1,
    g: 2,
  },
  assert std.isObject(self.object),
}
