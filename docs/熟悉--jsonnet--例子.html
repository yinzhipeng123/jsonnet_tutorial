  welcome: 'Hello ' + name + '!',

生成json


#vim test.jsonnet
local Person(name='Alice') = {
  name: name,
  welcome: 'Hello ' + name + '!',
};
{
  person1: Person(),
  person2: Person('Bob'),
}

# jsonnet test.jsonnet -o test.json
# cat test.json
{
   "person1": {
      "name": "Alice",
      "welcome": "Hello Alice!"
   },
   "person2": {
      "name": "Bob",
      "welcome": "Hello Bob!"
   }
}


生成多个json



#vim test.jsonnet
{
  "a.json": {
    x: 1,
    y: $["b.json"].y,
  },
  "b.json": {
    x: $["a.json"].x,
    y: 2,
  },
}

#jsonnet test.jsonnet -m ./
./a.json
./b.json

# cat a.json 
{
   "x": 1,
   "y": 2
}
# cat b.json 
{
   "x": 1,
   "y": 2
}
