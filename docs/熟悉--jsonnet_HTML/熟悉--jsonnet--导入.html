导入


import



用import可以导入 Jsonnet 代码。
为按惯例导入的文件一般为.libsonnet 结尾
原始 JSON 也可以通过这种方式导入。
其他文件也可以这么导入也可以，只要符合UTF-8

[root@centos7 jsonnet_learn]# cat garnish.txt 
Maraschino Cherry
[root@centos7 jsonnet_learn]# cat martinis.libsonnet 
{
  'Vodka Martini': "appple",
}
[root@centos7 jsonnet_learn]# cat test.jsonnet 
local martinis = import 'martinis.libsonnet';

{
  'Vodka Martini': martinis['Vodka Martini'],
    garnish: importstr 'garnish.txt',
  }
[root@centos7 jsonnet_learn]# jsonnet test.jsonnet 
{
   "Vodka Martini": "appple",
   "garnish": "Maraschino Cherry\n"
}
